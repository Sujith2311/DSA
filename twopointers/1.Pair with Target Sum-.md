class Solution(object):
    def twoSum(self, numbers, target):
        n=len(numbers)
        for i in range(n):
            need=target-numbers[i]
            left=i+1
            right=n-1
            while left<=right:
                mid=(left+right)//2
                if numbers[mid]==need:
                    return [i+1,mid+1]
                elif numbers[mid]<need:
                    left=mid+1
                else:
                    right=mid-1
